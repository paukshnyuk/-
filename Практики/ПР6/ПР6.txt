#include <8051.h>

void tput (unsigned char ch)
{
	SBUF = ch;
	while(!TI);
	TI = 0;
}

unsigned char getChar()
{
	unsigned char ch;
	while(!RI);
	ch = SBUF;
	tput(ch);
	RI = 0;
	return ch;
}
